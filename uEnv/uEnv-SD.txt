console=ttyO0,115200n8
loadaddr=0x82000000
fdtaddr=0x88000000
bootabspath=/BOOT
rootfspath=/dev/mmcblk0p2
bootargs=console=${console} root=${rootfspath} rootfstype=ext4 rootwait debug earlyprintk mem=512M
bootcmd=echo "Booting from microSD ..."; setenv autoload no; load mmc 0:1 ${loadaddr} ${bootabspath}/uImage; load mmc 0:1 ${fdtaddr} ${bootabspath}/am335x-boneblack.dtb; run bootargs; bootm ${loadaddr} - ${fdtaddr}
uenvcmd=run bootcmd
